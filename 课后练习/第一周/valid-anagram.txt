#leetcode:242
#给定两个字符串 s 和 t ，编写一个函数来判断 t 是否是 s 的字母异位词。

#解法一 字节数组排序

class Solution {
    public boolean isAnagram(String s, String t) {
        //字节数组排序
        if (s.length() != t.length()){
            return false;
        }
        char[] chars = s.toCharArray();
        char[] chart = t.toCharArray();
        Arrays.sort(chars);
        Arrays.sort(chart);
        return Arrays.equals(chars, chart);
    }
}
